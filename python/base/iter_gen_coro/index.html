    Simple decorator for initializing a generator

    def initgen(func):
    def wrapper(*args,**kwargs):
        gen = func(*args,**kwargs)
        next(gen)
        return gen
    return wrapper

    @initgen
def moyenne():
    _sum = 0
    _i = 0
    _moy = 0
    while True:
        elt = (yield f"Moyenne {_moy}")
        _sum += elt
        _i += 1
        _moy = _sum / _i

    def fibonacci(n):
    """ A generator for creating the Fibonacci numbers """
    a, b, counter = (0, 1, 0)
    while True:
        if (counter > n): 
            return
        yield a
        a, b = b, a + b
        counter += 1
f = fibonacci(5)
for x in f:
    print(x, "-> ", end="", flush=True)
